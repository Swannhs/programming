## C Implementation
### 14. Write a C program to accept customer details such as: Account_no, Name, Balance using structure. Assume 3 customers in the bank. Write a function to print the account no. and name of each customer whose balance < 100 Rs.
```c
//Write a C++ program to accept customer details such as: Account_no, Name, Balance using structure. Assume 3 customers in the bank. Write a function to print the account no. and name of each customer whose balance < 100 Rs.
#include<stdio.h>
struct account{
    char Name[50];
    long int Account_no;
    double Balance;
};
void inputdetails(struct account *a){
    printf("Name : ");
    fgets(a->Name,sizeof(a->Name),stdin);
    printf("account No : ");
    scanf("%ld",&a->Account_no);
    printf("enter balance:" );
    scanf("%lf",&a->Balance);
    getchar();
}
void lowBalanceCheck(struct account *a){
    if(a->Balance<100){
        printf("Account Name     : %s",a->Name);
        printf("Account No       : %ld\n",a->Account_no);
        printf("Account Balance  : %lf\n",a->Balance);
    }
}
int main(){
    struct account a,b,c;
    printf("Enter account details of first client\n");
    inputdetails(&a);
    printf("Enter account details of first client\n");
    inputdetails(&b);
    printf("Enter account details of first client\n");
    inputdetails(&c);
    printf("Low balance check : \n");
    lowBalanceCheck(&a);
    lowBalanceCheck(&b);
    lowBalanceCheck(&c);
}
```
